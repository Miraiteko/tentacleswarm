#确认目标
tentacleswarm_court.0001 = {
	
	hidden = yes
	
	trigger = {
		#有猎人
		employs_court_position = tentacle_hunter_court_position
		#有未知的奴隶
		any_courtier_or_guest = {
			is_tentacle_slave = yes
			is_imprisoned = no
			NOT = { has_court_position = tentacle_hunter_court_position }
			NOT = { is_tentacle_slave_known_by = { CHARACTER = ROOT } }
		}
	}	
	weight_multiplier = {  #首都腐化越高，越倾向于触发灭杀而不是这个
		base = 2
		modifier = { 
			factor = 0.5
			exists = capital_county
			capital_county.var:tentacleswarm_corruption  = { compare_value > 35 }
		}
		modifier = { 
			factor = 0
			exists = capital_county
			capital_county.var:tentacleswarm_corruption  = { compare_value > 60 }
		}
	}	
	immediate = {
		#设定目标
		random_court_position_holder = {
			type = tentacle_hunter_court_position
			save_scope_as = hunter
		}
		random_courtier_or_guest = {
			limit = {
				is_tentacle_slave = yes
				is_imprisoned = no
				NOT = { has_court_position = tentacle_hunter_court_position }
				NOT = { is_tentacle_slave_known_by = { CHARACTER = ROOT } }
			}
			save_scope_as = target
		}
		#双方进行密谋对抗，如果成功，则直接发现秘密
		scope:hunter = {
			duel = { #技能对抗，初始30%成功率，每比对方高2点密谋，增加10%成功率，最高90%
				skill = intrigue
				target = scope:target

				3 = {
					trigger = {
						scope:hunter  = {
							is_tentacle_slave = no #如果猎人是奴隶的话，不会直接揪出来
						}
					}
					compare_modifier = {
						value = scope:duel_value
						multiplier = 0.5
						min = -3
					}
					ROOT = { trigger_event = tentacleswarm_court.0002 }
				}

				7 = {
					compare_modifier = {
						value = scope:duel_value
						multiplier = -0.5
						min = -6
					}
					ROOT = { trigger_event = tentacleswarm_court.0003 }
				}
			}
		}
	}
}

tentacleswarm_court.0011 = {
	
	hidden = yes
	
	trigger = {
		#有猎人
		any_court_position_holder = {
			type = tentacle_hunter_court_position
			is_tentacle_slave = no
		}
		#有筑巢省份
		any_held_title = {
			tier = tier_county
			var:tentacleswarm_corruption = { compare_value > 35 }
		}
	}
	weight_multiplier = {  #首都腐化越高，越倾向于触发这个
		base = 1
		modifier = { 
			factor = 2
			exists = capital_county
			capital_county.var:tentacleswarm_corruption  = { compare_value > 60 }
		}
		modifier = { 
			factor = 4
			exists = capital_county
			capital_county.var:tentacleswarm_corruption  = { compare_value > 100 }
		}
	}
	immediate = {
		#猎人
		random_court_position_holder = {
			type = tentacle_hunter_court_position
			save_scope_as = hunter
		}
		#将腐化最高的省份设为目标
		ordered_held_title = {
			limit = {
				tier = tier_county
				var:tentacleswarm_corruption = { compare_value > 35 }
			}
			order_by = var:tentacleswarm_corruption
			save_scope_as = target_county
		}
		trigger_event = tentacleswarm_court.0012
	}
}
