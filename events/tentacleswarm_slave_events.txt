	desc = tentacleswarm_slave.2002.desc
	theme = seduction

	widget = {
		gui = "event_window_widget_scheme"

	}
	trigger = {
		is_female = yes
		fertility >= 0.1
		is_tentacle_body_corrupted = no
	}
	option = {
		name = tentacleswarm_slave.2002.a
		tentacle_body_corrupt = yes
		#立刻去寻求赐福
		trigger_event = {
			id = tentacleswarm_slave.0031

		}
	}
	option = {
		name = tentacleswarm_slave.2002.b
		tentacle_body_corrupt = yes
		#忍受，增加压力
		add_stress = massive_stress_gain
	}
	after = {
		#下一次成瘾事件
		trigger_event = { 
			id = tentacleswarm_slave.3001
			months = { 2 12 }
		}
	}
}
#成瘾性事件，定期需要去接受赐福，否则增加压力
tentacleswarm_slave.3001 = { #预处理事件，怀孕的奴隶角色延迟触发
	hidden = yes
	immediate = {
		if = {
			limit = { 
				is_pregnant = yes
				is_tentacle_slave = yes
			}
			trigger_event = { 
				id = tentacleswarm_slave.3001
				months = 10
			}
		}
		else = {
			trigger_event = { 
				id = tentacleswarm_slave.3002
				days = 1
			}
		}
	}
}
