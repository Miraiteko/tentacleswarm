#1001 奴隶发动政变
namespace = tentacle_slave_claim_throne

#奴隶发动政变
tentacle_slave_claim_throne.1001 = {
	hidden = yes
	immediate = {
		#计算被发现概率
		save_scope_value_as = {
			name = discovery_chance
			value = {
				value = 100
				subtract = scope:scheme.scheme_secrecy
			}
		}
		#判断是否被发现
		random = {
			chance = scope:discovery_chance
			custom_tooltip = claim_throne_become_discovered_roll_tt
			save_scope_value_as = {
				name = scheme_discovered
				value = yes
			}
		}
		#判断是否成功
		random = {
			chance = scope:scheme.scheme_success_chance
			custom_tooltip = claim_throne_successful_roll_tt
			save_scope_value_as = {
				name = scheme_successful
				value = yes
			}
		}
		#成功效果
		if = {
			limit = {
				exists = scope:scheme_successful
			}
			scope:target = { #玩家被夺位的话，扮演夺位者
				if = {
					limit = { is_ai = no }
					set_player_character = scope:owner
				}
			}
			scope:owner = {
				#获得目标头衔
				create_title_and_vassal_change = {
					type = usurped
					save_scope_as = change
					add_claim_on_loss = yes
				}
				scope:target = {
					every_held_title = {
						limit = {
							tier >= tier_county
						}
						add_to_list = target_titles
					}
				}
				setup_claim_cb = {
					titles = target_titles
					attacker = scope:owner
					defender = scope:target
					claimant = scope:owner
					change = scope:change
				}	
				resolve_title_and_vassal_change = scope:change
				#改信
				convert_to_tentacleswarm_religion = yes
				#释放费洛蒙 
				tentacleswarm_4VA = yes
			}
		}
		#没成功，被发现了
		else_if = {
			limit = {
				exists = scope:scheme_discovered
			}
			scope:target = {
				imprison = {
					target = scope:owner
					type = dungeon
				}
			}
			scope:owner = {
				random_secret = {
					limit = { secret_type = secret_tentacle_slave }
					expose_secret = scope:target
				}
			}
		}
		else = { #没成功，但是没被发现
		}
		
		#结束密谋
		scope:scheme = {
			end_scheme = yes
		}
	}
}
