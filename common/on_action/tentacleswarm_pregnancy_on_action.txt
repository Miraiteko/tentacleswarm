#怀孕触发
on_pregnancy_mother = { #define设置，怀孕之后280天出生，60天时知道怀孕
	on_actions = {
		tentacleswarm_on_pregnancy_pulse
	}
}
tentacleswarm_on_pregnancy_pulse = { 
	on_actions = {
		tentacleswarm_baby_corruption #主人赐福
		delay = { days = 200 }
		tentacleswarm_near_birth_pulse #接近出生时触发
	}
}
tentacleswarm_baby_corruption = {
	trigger = { #目前只考虑奴隶被赐福
		is_tentacle_slave = yes
		is_imprisoned = no #不能被囚禁
		OR = {
			is_ruler = yes #或者是一位领主，可以自由地寻找机会去寻找主人
			any_spouse = { is_ruler = yes } #或者是一位领主夫人，会尽可能地找机会腐化未来的继承人
			has_trait = tentacle_slave #或者是已经公开的奴隶
		}
	}
	effect = {
		add_character_flag = { flag = tentacle_baby_corruption months = 10 } 
	}
}
tentacleswarm_near_birth_pulse = {
	trigger = { is_pregnant = yes } #如果提前流产了的话，就不用考虑了 
	effect = {
		#天生奴遗传确认，该性状为伴X隐形基因，雄性不表现
		
		#母亲为纯合子
		if = { 
			limit = {
				has_character_flag = congenital_tentacle_slave
			}
			#母亲被赐福
			if = { 
				limit = { 
					has_character_flag = tentacle_baby_corruption
				}
				#孩子一定是女性纯合
				set_pregnancy_gender = female 
				add_character_flag = { flag = baby_congenital_tentacle_slave months = 5 } 
			}
			#父亲活着，是携带者
			else_if = {
				limit = { 
					scope:real_father = { 
						is_alive = yes
						has_character_flag = congenital_tentacle_slave_carry
					} 
				}
				random_list = {
					50 = {#女性纯合
						set_pregnancy_gender = female
						add_character_flag = { flag = baby_congenital_tentacle_slave months = 5 }
					}
					50 = {#男性携带
						set_pregnancy_gender = male
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
				}
			}
			#父亲是正常人
			else = {
				random_list = {
					50 = {#女性携带
						set_pregnancy_gender = female
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
					50 = {#男性携带
						set_pregnancy_gender = male
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
				}
			}
		}
		
		#母亲为携带者
		else_if = { 
			limit = {
				has_character_flag = congenital_tentacle_slave_carry
			}
			#母亲被赐福
			if = { 
				limit = { 
					has_character_flag = tentacle_baby_corruption
				}
				random_list = {
					50 = {#女性纯合
						set_pregnancy_gender = female
						add_character_flag = { flag = baby_congenital_tentacle_slave months = 5 }
					}
					50 = {#女性携带
						set_pregnancy_gender = female
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
				}
			}
			#父亲是携带者
			else_if = {
				limit = { 
					scope:real_father = { has_character_flag = congenital_tentacle_slave_carry } 
				}
				random_list = {
					25 = {#女性纯合
						set_pregnancy_gender = female
						add_character_flag = { flag = baby_congenital_tentacle_slave months = 5 }
					}
					25 = {#女性携带
						set_pregnancy_gender = female
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
					25 = {#男性携带
						set_pregnancy_gender = male
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
					25 = {#男性正常
						set_pregnancy_gender = male
					}
				}
			}
			#父亲是正常人
			else = {
				random_list = {
					25 = {#女性携带
						set_pregnancy_gender = female
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
					25 = {#女性正常
						set_pregnancy_gender = female
					}
					25 = {#男性携带
						set_pregnancy_gender = male
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
					25 = {#男性正常
						set_pregnancy_gender = male
					}
				}
			}
		}
		#母亲为正常人
		else = { 
			#母亲被赐福
			if = { 
				limit = { 
					has_character_flag = tentacle_baby_corruption 
				}
				#孩子一定是女性携带
				set_pregnancy_gender = female 
				add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 } 
			}
			#父亲是携带者
			else_if = {
				limit = { 
					scope:real_father = { has_character_flag = congenital_tentacle_slave_carry } 
				}
				random_list = {
					50 = {#女性携带
						set_pregnancy_gender = female
						add_character_flag = { flag = baby_congenital_tentacle_slave_carry months = 5 }
					}
					50 = {#男性正常
						set_pregnancy_gender = male
					}
				}
			}
			#父亲是正常人
			else = {
				#孩子正常
			}
		}
	}
}


#出生触发
on_birth_child = {
	on_actions = {
		tentacleswarm_birth_pulse
	}
}
#TS出生结算
tentacleswarm_birth_pulse = {
	effect = {
		#新生儿天生奴遗传结算
		if = { #天生奴隶
			limit = {
				is_female = yes
				mother = { has_character_flag = baby_congenital_tentacle_slave } 
			}
			add_character_flag = congenital_tentacle_slave #成为纯合子
			add_trait = congenital_tentacle_slave #成为奴隶，测试
		}
		else_if = { #有的时候有bug让男孩成为纯合子，改为成为携带者
			limit = {
				is_female = no
				mother = { has_character_flag = baby_congenital_tentacle_slave } 
			}
			add_character_flag = congenital_tentacle_slave_carry #成为携带者
			add_trait = congenital_tentacle_slave_carry #成为奴隶，测试
		}	
		else_if = { #携带者
			limit = {
				mother = { has_character_flag = baby_congenital_tentacle_slave_carry } 
			}
			add_character_flag = congenital_tentacle_slave_carry #成为携带者
			add_trait = congenital_tentacle_slave_carry #成为奴隶，测试
		}	
		else = { #正常人
			#add_trait = congenital_tentacle_slave_no #成为奴隶，测试
		}
	}
}
#接近成年
on_15th_birthday = {
	on_actions = {
		tentacleswarm_adult_pulse
	}
}
tentacleswarm_adult_pulse = {
	effect = {
		if = { #天生奴隶成年时成为奴隶
			limit = {
				is_female = yes
				has_character_flag = congenital_tentacle_slave 
			}
			become_tentacle_slave = yes
		}
		else_if = { #其他女性看成年所在地的腐化度
			limit = { is_female = yes }
			random_list = {
				0 = {
					modifier = {
						add = location.county.var:tentacleswarm_corruption
						exists = location
						exists = location.county
					}
					modifier = { #监护人是奴隶，权重+50
						add = 50
						any_relation = {
							type = guardian
							is_tentacle_slave = yes
						}
					}
					become_tentacle_slave = yes
				}
				100 = {
					modifier = {
						add = location.county.minus_tentacleswarm_corruption
						exists = location
						exists = location.county
					}
				}
			}
		}
	}
}
