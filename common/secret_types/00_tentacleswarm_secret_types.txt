#只能通过"become_tentacle_slavery"来添加
secret_tentacle_slavery = {
	category = deviancy

	is_valid = {
		scope:secret_owner = {
			NOT = {
				has_trait = tentacle_slavery #如果已经公开了，就无效了
			}
		}
	}

	is_shunned = { #以弱牵制勒索的条件
		scope:secret_owner = { secret_tentacle_slavery_is_shunned_trigger = yes}
	}
	is_criminal = {	#以强牵制勒索的条件
		OR = {
			scope:secret_owner = { secret_tentacle_slavery_is_criminal_trigger = yes}
			scope:secret_owner = { secret_tentacle_slavery_is_evil_trigger = yes}
		}
	}

	
	on_expose = {
		save_scope_as = secret
		
		scope:secret_owner = {
			if = {
				limit = {
					NOT = {
						this = scope:secret_exposer
					}
				}
				secret_exposed_owner_opinion_effects_effect = yes  #憎恨揭露秘密的人
				play_music_cue = "mx_cue_secret"
			}
		}
		#成为公开的奴隶
		scope:secret_owner = {
			become_openly_tentacle_slavery_effect = {
					SECRET_EXPOSE = yes #揭露秘密导致的提升
					EXPOSER =  scope:secret_exposer #揭露者
			}
		}
	}
}
