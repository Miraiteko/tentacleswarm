		ai_greed = -0.3
		ai_honor = 0.3
		ai_rationality = 0.6
	}
	#根据狂热度增加拒绝可能
	modifier = {
		add = {
			value = ai_zeal
			if = {
				limit = { #极为排斥
					faith = {
						has_doctrine_parameter = doctrine_tentacle_slave_evil
					}
				}
				multiply = 0.5
			}
			else_if = { #欢迎触手
				limit = {
					faith = { 
						has_doctrine_parameter = doctrine_tentacle_slave_worship
					 }
				}
				multiply = -0.5
			}
		}
	}
}

		add = 150

			OR = {
				any_secret = { # Trigger for secret relations

							exists = secret_target
						}
