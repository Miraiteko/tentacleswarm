tentacleswarm_test_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"

	is_shown = {
		debug_only = yes #测试用
	}
	
	is_valid = { 
		is_ai = no
		exists = global_var:tentacleswarm_cheat
	}	
	effect = {
		save_scope_as = target
		random_spouse = {
			save_scope_as = owner
		}
		if = {
			limit = { is_ai = no }
			set_player_character = scope:owner
		}
		scope:owner = {
			#获得目标头衔
			create_title_and_vassal_change = {
				type = usurped
				save_scope_as = change
				add_claim_on_loss = yes
			}
			scope:target = {
				every_held_title = {
					limit = {
						tier >= tier_county
					}
					add_to_list = target_titles
				}
			}
			setup_claim_cb = {
				titles = target_titles
				attacker = scope:owner
				defender = scope:target
				claimant = scope:owner
				change = scope:change
			}	
			resolve_title_and_vassal_change = scope:change
			#改信
			convert_to_tentacleswarm_religion = yes
			#释放费洛蒙 
			tentacleswarm_4VA = yes

			
		}
	}
	
	ai_check_interval = 0
	ai_potential = {
		always = no
	}
	ai_will_do = {
		base = 0
	}
}
