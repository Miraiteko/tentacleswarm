tentacleswarm_start_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"

	desc = tentacleswarm_start_decision_desc
	selection_tooltip = tentacleswarm_start_decision_tooltip

	is_shown = {
		is_ai = no  #ai不通过这个来皈依
		any_secret = { secret_type = secret_tentacle_slave }
	}
	
	is_valid = { 
		any_secret = { secret_type = secret_tentacle_slave }
	}	
	effect = {
		#公开皈依
		convert_to_tentacleswarm_religion = yes
		#信号施放
		tentacleswarm_4VA = yes
	}
	
	ai_check_interval = 60

	ai_potential = {
		OR = {
			NOT = { exists = capital_province }
			capital_province.county = { #首都可以召出5队巨兽，ai才会点
				var:tentacleswarm_corruption_warbeast = { compare_value >= 500 }
			}
		}
	}

	ai_will_do = {
		base = 0
	}
}

tentacleswarm_test_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"

	desc = tentacleswarm_start_decision_desc
	selection_tooltip = tentacleswarm_start_decision_tooltip

	is_shown = {
		is_ruler = yes #统治者
	}
	
	is_valid = { 
		is_ai = no
	}	
	effect = {
		add_gold =100
		set_variable =  { name = max_held_development value = 1 }
		every_held_title = {
			limit = { tier = tier_county }
			if = {
				limit = {
					development_level > holder.var:max_held_development
				}
				holder = { set_variable =  { name = max_held_development value = prev.development_level }}
			}
		}
		change_variable = { name = max_held_development multiply = 2 } #目标地要是原来的两倍以上发展度
	}
	
	ai_check_interval = 60

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
