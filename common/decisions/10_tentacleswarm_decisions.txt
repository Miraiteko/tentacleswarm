tentacleswarm_start_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds" 

	desc = tentacleswarm_start_decision_desc
	selection_tooltip = tentacleswarm_start_decision_tooltip

	is_shown = {
		is_ai = no  #ai不通过这个来皈依
		not = { has_trait = tentacle_slave }
		any_secret = { secret_type = secret_tentacle_slave }
	}
	
	is_valid = {
		any_secret = { secret_type = secret_tentacle_slave }
	}	
	effect = {
		#公开皈依
		convert_to_tentacleswarm_religion = yes  
		#信号施放
		tentacleswarm_4VA = yes
	}
	
	ai_check_interval = 60

	ai_potential = {
		OR = {
			NOT = { exists = capital_province }
			capital_province.county = { #首都可以召出5队巨兽，ai才会点
				var:tentacleswarm_corruption_warbeast = { compare_value >= 500 }
			}
		}
	}

	ai_will_do = {
		base = 0
	}
}

tentacleswarm_test_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"

	is_shown = {
		debug_only = yes #测试用
	}
	
	is_valid = { 
		is_ai = no
		debug_only = yes
	}	
	effect = {
		capital_county = {
			every_county_province = { #通过变换地产类型来清建筑
				if = {
					limit= {
						has_holding_type = castle_holding
					}
					set_holding_type = tribal_holding
					set_holding_type = castle_holding
				}
				else_if = {
					limit= {
						has_holding_type = tribal_holding
					}
					set_holding_type = city_holding
					set_holding_type = tribal_holding
				}
				else_if = {
					limit= {
						has_holding_type = city_holding
					}
					set_holding_type = tribal_holding
					set_holding_type = city_holding
				}
				else_if = {
					limit= {
						has_holding_type = church_holding
					}
					set_holding_type = tribal_holding
					set_holding_type = church_holding
				}
				else_if = {
					limit= {
						has_holding = yes
					}
					set_holding_type = tribal_holding
				}
			}
		}
	}
	
	ai_check_interval = 0
	ai_potential = {
		always = no
	}
	ai_will_do = {
		base = 0
	}
}
