tentacleswarm_start_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"

	desc = tentacleswarm_start_decision_desc
	selection_tooltip = tentacleswarm_start_decision_tooltip

	is_shown = {
		is_ruler = yes #统治者
		any_secret = { secret_type = secret_tentacle_slave }
	}
	
	is_valid = { 
		any_secret = { secret_type = secret_tentacle_slave }
	}	
	effect = {
		#公开皈依
		convert_to_tentacleswarm_religion = yes
		#廷臣皈依
		every_courtier_or_guest	= {
			limit = {
				is_tentacle_slave = yes
			}
			convert_to_tentacleswarm_religion = yes
		}
		#封臣皈依
		every_vassal_or_below ={
			if = {
				limit = { is_tentacle_slave = yes } 
				convert_to_tentacleswarm_religion = yes
			}
			every_courtier_or_guest	= {
				limit = {
					is_tentacle_slave = yes
				}
				convert_to_tentacleswarm_religion = yes
			}
		}
	}
	
	ai_check_interval = 60

	ai_potential = {
		OR = {
			NOT = { exists = capital_province }
			capital_province.county = { #首都可以召出5队巨兽，ai才会点
				var:tentacleswarm_corruption_warbeast = { compare_value >= 500 }
			}
		}
	}

	ai_will_do = {
		base = 100
	}
}
tentacleswarm_test_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"

	desc = tentacleswarm_start_decision_desc
	selection_tooltip = tentacleswarm_start_decision_tooltip

	is_shown = {
		is_ruler = yes #统治者
	}
	
	is_valid = { 
		is_ai = no
		basic_is_valid_for_yearly_events_trigger = yes 
		has_religion = religion:tentacleswarm_religion
		#is_ai = yes
		is_at_war = no
		max_military_strength > 3000
		piety > 50
	}	
	effect = {
			save_scope_as = actor
			
			random_independent_ruler = {
				limit = {
					is_landed = yes #有地
					#NOT = { is_allied_to = ROOT } #不能宣战自己的盟友
					NOT = { has_religion = religion:tentacleswarm_religion } #异教
					capital_county = {  #首都有发展度
						development_level > 0
					}
					#ROOT = { #需要符合宣战条件
					#	is_character_interaction_valid = {
					#	   recipient = PREV
					#	   interaction = declare_war_interaction
					#	}
					#}
				}
				save_scope_as = target_character
				save_scope_as = recipient
				if = {
					limit = {
						war_declarer_needs_hook_on_liege = no #不需要领主同意就能发动战争
					}
					scope:actor = {
						start_war = { 
							casus_belli = tentacleswarm_conquest 
							target = scope:target_character
							target_title = scope:target_character.capital_county.kingdom
						}
					}
				}
			}
	}
	
	ai_check_interval = 60

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
